Create Relative Time Variables

# Filter to only include treatment repositories (those that adopted Cursor)
panel_data <- panel_data %>%
    filter(event > 0) %>%  # Only repositories that actually adopted Cursor
    arrange(repo_name, time) %>%
    group_by(repo_name) %>%
    mutate(
        relative_time = time - event
    ) %>%
    ungroup() %>%
    # Filter to only include observations within -6 to +6 months of event
    filter(relative_time >= -6 & relative_time <= 6)

library(kableExtra)

# Calculate mean outcomes for each relative time point
means_by_time <- panel_data %>%
    group_by(relative_time) %>%
    summarise(
        n = n(),
        across(all_of(outcome_vars), ~ mean(.x, na.rm = TRUE)),
        .groups = "drop"
    )

# Calculate overall pre vs post means
pre_post_summary <- panel_data %>%
    mutate(period = ifelse(relative_time < 0, "Pre", "Post")) %>%
    group_by(period) %>%
    summarise(
        n = n(),
        across(all_of(outcome_vars), ~ mean(.x, na.rm = TRUE)),
        .groups = "drop"
    ) %>%
    arrange(desc(period))  # Pre first, then Post

# Calculate difference (Post - Pre)
pre_vals <- pre_post_summary %>% filter(period == "Pre")
post_vals <- pre_post_summary %>% filter(period == "Post")

# Build difference row
diff_row <- tibble(period = "Difference (Post - Pre)", n = NA_integer_)
for (var in outcome_vars) {
    diff_row[[var]] <- post_vals[[var]] - pre_vals[[var]]
}

# Build percent change row
pct_row <- tibble(period = "% Change", n = NA_integer_)
for (var in outcome_vars) {
    pct_row[[var]] <- 100 * (post_vals[[var]] - pre_vals[[var]]) / pre_vals[[var]]
}

# Perform Wilcoxon rank-sum tests for each outcome
pval_row <- tibble(period = "p-value (Wilcoxon)", n = NA_integer_)
for (var in outcome_vars) {
    pre_data <- panel_data %>% filter(relative_time < 0) %>% pull(!!sym(var))
    post_data <- panel_data %>% filter(relative_time >= 0) %>% pull(!!sym(var))
    test <- wilcox.test(post_data, pre_data)
    pval_row[[var]] <- test$p.value
}

Interrupted Time Series Analysis

library(fixest)

# Prepare ITS variables
# time: sequential time (0 to 12 for months -6 to +6)
# post: indicator for post-intervention (relative_time >= 0)
# time_after: time since intervention (0 before, 1,2,3... after)

its_data <- panel_data %>%
    mutate(
        time_seq = relative_time + 7,  # Convert -6 to +6 → 1 to 13
        post = ifelse(relative_time >= 0, 1, 0),
        time_after = ifelse(relative_time >= 0, relative_time + 1, 0)
    )

# Helper function to add significance stars
add_stars <- function(estimate, pval) {
    stars <- case_when(
        pval < 0.001 ~ "***",
        pval < 0.01 ~ "**",
        pval < 0.05 ~ "*",
        TRUE ~ ""
    )
    paste0(round(estimate, 4), stars)
}

# Helper to format estimate with SE in parentheses (with line break)
format_coef <- function(est, se, pval) {
    paste0(add_stars(est, pval), "<br>(", round(se, 4), ")")
}

# Build summary table combining all four ITS models
summary_results <- list()

for (i in seq_along(outcome_vars)) {
    var <- outcome_vars[i]
    
    # Model 1: No covariates, no FE
    formula1 <- as.formula(paste(var, "~ time_seq + post + time_after"))
    model1 <- lm(formula1, data = its_data)
    coefs1 <- coef(summary(model1))
    
    # Model 2: With covariates, no FE
    formula2 <- as.formula(paste(var, "~ time_seq + post + time_after + age + ncloc + contributors + stars + issues"))
    model2 <- lm(formula2, data = its_data)
    coefs2 <- coef(summary(model2))
    
    # Model 3: Repo FE only (no covariates)
    formula3 <- as.formula(paste(var, "~ time_seq + post + time_after | repo_name"))
    model3 <- feols(formula3, data = its_data)
    coefs3 <- coeftable(model3)
    
    # Model 4: Repo FE + covariates
    formula4 <- as.formula(paste(var, "~ time_seq + post + time_after + age + ncloc + contributors + stars + issues | repo_name"))
    model4 <- feols(formula4, data = its_data)
    coefs4 <- coeftable(model4)
    
    summary_results[[i]] <- tibble(
        Outcome = outcome_names[i],
        N = format(nobs(model1), big.mark = ","),
        # No covariates, no FE
        `Pre` = format_coef(coefs1["time_seq", "Estimate"], coefs1["time_seq", "Std. Error"], coefs1["time_seq", "Pr(>|t|)"]),
        `Level` = format_coef(coefs1["post", "Estimate"], coefs1["post", "Std. Error"], coefs1["post", "Pr(>|t|)"]),
        `Post` = format_coef(coefs1["time_after", "Estimate"], coefs1["time_after", "Std. Error"], coefs1["time_after", "Pr(>|t|)"]),
        `R²` = round(summary(model1)$r.squared, 3),
        # With covariates, no FE
        `Pre ` = format_coef(coefs2["time_seq", "Estimate"], coefs2["time_seq", "Std. Error"], coefs2["time_seq", "Pr(>|t|)"]),
        `Level ` = format_coef(coefs2["post", "Estimate"], coefs2["post", "Std. Error"], coefs2["post", "Pr(>|t|)"]),
        `Post ` = format_coef(coefs2["time_after", "Estimate"], coefs2["time_after", "Std. Error"], coefs2["time_after", "Pr(>|t|)"]),
        `R² ` = round(summary(model2)$r.squared, 3),
        # Repo FE only
        `Pre  ` = format_coef(coefs3["time_seq", "Estimate"], coefs3["time_seq", "Std. Error"], coefs3["time_seq", "Pr(>|t|)"]),
        `Level  ` = format_coef(coefs3["post", "Estimate"], coefs3["post", "Std. Error"], coefs3["post", "Pr(>|t|)"]),
        `Post  ` = format_coef(coefs3["time_after", "Estimate"], coefs3["time_after", "Std. Error"], coefs3["time_after", "Pr(>|t|)"]),
        `R²  ` = round(r2(model3, type = "ar2"), 3),
        # Repo FE + covariates
        `Pre   ` = format_coef(coefs4["time_seq", "Estimate"], coefs4["time_seq", "Std. Error"], coefs4["time_seq", "Pr(>|t|)"]),
        `Level   ` = format_coef(coefs4["post", "Estimate"], coefs4["post", "Std. Error"], coefs4["post", "Pr(>|t|)"]),
        `Post   ` = format_coef(coefs4["time_after", "Estimate"], coefs4["time_after", "Std. Error"], coefs4["time_after", "Pr(>|t|)"]),
        `R²   ` = round(r2(model4, type = "ar2"), 3)
    )
}
